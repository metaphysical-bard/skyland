    これ is 何？ #

    skylandはC++AMPのようにCPUとGPUの処理を同じ文法ルールで記述することができるプログラミング言語です。


    Example Code #

    //main function
main() {
  log("Hello World!")
}

    //add
main() {
  let a = 10
  let b = 10
  let c = add(a, b)
  log(c)
}

    func(a) {
  return add(a, 1)
}

main() {
  let a = {0, 1, 2, 3}//array type
  let b = {0, 0, 0, 0}
  let c = 0

  /* by the CPU */
  log("CPU : ")
  for (let i, 4) {
    c = func(a[i])
    log(c)
  }

  log("\n")

  /* by the GPU */
  log("GPU")
  ga = setBuf("DST", a)
  gb = setBuf("SRC")
  gb = parallel_for_each(4, 1, 1, func, ga)
  copy(gb, b)
  for (let i, 4) {
    log(b[i])
  }
}
