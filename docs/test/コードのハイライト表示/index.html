<script type="text/javascript">
    let list = document.body.getElementsByClassName("language-testcode")

    for (let i = 0; i < list.length; i++) {
        let str = list[i].innerText + "\n"
        list[i].innerHTML = str
            .replace(/test1/gu, "<span style=\"color:#d44336\">test</span>")
            .replace(/test2/gu, "<span style=\"color:#66d9ef\">test</span>")
            .replace(/test3/gu, "<span style=\"color:#ebc02d\">test</span>")
            .replace(/test4/gu, "<span style=\"color:#a6e22e\">test</span>")
    }
</script>
